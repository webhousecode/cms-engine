# F48 — Internationalization (i18n)

> Complete multi-language content management with translation workflows, side-by-side editing, AI translation, locale routing helpers, and a language switcher component.

## Problem

The storage layer already supports `locale` and `translationOf` fields on documents, and `cms.config.ts` accepts `defaultLocale` and `locales`. But the admin UI has no translation workflow — editors can't see translations side-by-side, can't initiate translations, and there are no routing helpers for frameworks. AI translation exists in `@webhouse/cms-ai` but isn't wired into the editor UI.

## Solution

Build the admin UI layer on top of the existing storage infrastructure. Side-by-side translation editor, translation status tracking, AI translation trigger from the editor, locale routing helpers for Next.js/Astro, and a drop-in language switcher component for sites.

### 6. Stale Translation Detection

```typescript
// packages/cms/src/i18n/stale.ts

export function isTranslationStale(source: Document, translation: Document): boolean {
  if (!source.updatedAt || !translation.updatedAt) return false;
  return new Date(source.updatedAt) > new Date(translation.updatedAt);
}
```

export function generateI18nStaticParams(
  collection: string,
  locales: string[],
): { locale: string; slug: string }[] {
  const params: { locale: string; slug: string }[] = [];
  for (const locale of locales) {

### 8. Language Switcher Component

A React component sites can drop into their layout:

```typescript
// packages/cms/src/components/language-switcher.tsx

"use client";
import { usePathname } from "next/navigation";

export function LanguageSwitcher({
  locales,
  labels,
}: {
  locales: string[];
  labels?: Record<string, string>;
}) {
  const pathname = usePathname();
  const DEFAULT_LABELS: Record<string, string> = {
    en: "English", da: "Dansk", de: "Deutsch", fr: "Francais",
    es: "Espanol", sv: "Svenska", nb: "Norsk", nl: "Nederlands",
  };

  function switchLocale(targetLocale: string) {
    const segments = pathname.split("/");
    if (locales.includes(segments[1])) {
      segments[1] = targetLocale;
    } else {
      segments.splice(1, 0, targetLocale);
    }
    return segments.join("/");
  }

  return (
    <nav aria-label="Language">
      {locales.map(locale => (
        <a key={locale} href={switchLocale(locale)}>
          {labels?.[locale] ?? DEFAULT_LABELS[locale] ?? locale}
        </a>
      ))}
    </nav>
  );
}
```
