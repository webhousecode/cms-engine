# F65 — Agent Pipeline E2E Tests

> Playwright end-to-end tests covering the full AI agent lifecycle: configure → schedule → run → curation queue → approve/reject → cockpit monitoring.

## Problem

The AI agent pipeline is the most complex and business-critical flow in the CMS:

```
Cockpit settings → Agent config → Schedule/Run Now → LLM call → Curation Queue → Approve/Reject → Published document
```

**Current test coverage: near zero.** Only 3 Playwright tests exist in `cms-admin/e2e/`, and they test superficial UI rendering (dropdowns, no hydration errors). None of them:

- Run an actual agent (even with mocked LLM)
- Test the curation queue approve/reject flow
- Verify scheduled agent execution
- Test cockpit parameter changes affecting agent behavior
- Verify cost tracking and budget enforcement
- Test the full roundtrip: agent generates → curation shows item → approve → document exists

If this pipeline breaks silently, AI-generated content stops flowing — and nobody knows until a customer complains. This is a ship blocker.

## Solution

Comprehensive Playwright E2E test suite covering every stage of the agent pipeline. Uses a **mocked LLM** (intercept Anthropic API calls, return deterministic responses) so tests are fast, free, and reproducible. Tests both the happy path and edge cases (budget exceeded, agent disabled, rejection feedback loop).

### Mock LLM Fixture

Intercept all Anthropic API calls so tests don't hit real LLM and cost $0:

```typescript
// packages/cms-admin/e2e/fixtures/mock-llm.ts

import { Page } from "@playwright/test";

/** Sample agent output matching CMS content format */
const MOCK_AGENT_RESPONSE = {
  id: "msg_test",
  type: "message",
  role: "assistant",
  content: [{
    type: "text",
    text: JSON.stringify({
      title: "Test Generated Article",
      slug: "test-generated-article",
      excerpt: "This is a test article generated by the mock LLM.",
      body: "<p>Mock content for E2E testing.</p>",
      author: "AI Agent",
      status: "draft",
    }),
  }],
  model: "claude-sonnet-4-6-20250514",
  stop_reason: "end_turn",
  usage: { input_tokens: 100, output_tokens: 200 },
};

export async function mockLlmResponses(page: Page) {
  await page.route("**/api.anthropic.com/**", async (route) => {
    await route.fulfill({
      status: 200,
      contentType: "application/json",
      body: JSON.stringify(MOCK_AGENT_RESPONSE),
    });
  });

  // Also mock OpenAI if multi-model is enabled
  await page.route("**/api.openai.com/**", async (route) => {
    await route.fulfill({
      status: 200,
      contentType: "application/json",
      body: JSON.stringify({
        choices: [{ message: { content: JSON.stringify({
          title: "Test Article (GPT)",
          slug: "test-article-gpt",
          body: "<p>Mock GPT content.</p>",
        }) } }],
        usage: { prompt_tokens: 100, completion_tokens: 200 },
      }),
    });
  });
}
```

### Test Helpers

```typescript
// packages/cms-admin/e2e/fixtures/test-helpers.ts

import { Page, expect } from "@playwright/test";

export async function loginAsAdmin(page: Page) {
  await page.goto("/admin/login");
  await page.fill('input[type="email"]', "cb@webhouse.dk");
  await page.fill('input[type="password"]', process.env.TEST_PASSWORD ?? "test1234");
  await page.click('button[type="submit"]');
  await page.waitForURL("**/admin**");
}

export async function navigateToAgents(page: Page) {
  await page.goto("/admin/agents");
  await page.waitForSelector('[data-slot="sidebar"]');
}

export async function navigateToCuration(page: Page) {
  await page.goto("/admin/curation");
}

export async function navigateToCockpit(page: Page) {
  await page.goto("/admin/command");
}

export async function runAgentNow(page: Page, agentId: string, prompt: string) {
  const res = await page.request.post(`/api/cms/agents/${agentId}/run`, {
    data: { prompt },
  });
  expect(res.ok()).toBeTruthy();
  return res.json();
}

export async function waitForQueueItem(page: Page, title: string, timeout = 10000) {
  await page.goto("/admin/curation");
  await expect(page.getByText(title)).toBeVisible({ timeout });
}
```

### Key Test Scenarios

#### 1. Full Roundtrip (most critical test)

```typescript
// packages/cms-admin/e2e/pipeline/full-roundtrip.spec.ts

test("agent run → curation queue → approve → document created", async ({ page }) => {
  await mockLlmResponses(page);
  await loginAsAdmin(page);

  // 1. Run agent via API
  const result = await runAgentNow(page, "content-writer", "Write a test article about testing");
  expect(result.queueItemId).toBeTruthy();

  // 2. Verify item appears in curation queue
  await navigateToCuration(page);
  await expect(page.getByText("Test Generated Article")).toBeVisible();
  expect(await page.getByText("ready").count()).toBeGreaterThan(0);

  // 3. Approve the item
  await page.getByText("Test Generated Article").click();
  await page.getByRole("button", { name: /approve/i }).click();

  // 4. Verify document was created in collection
  await page.goto("/admin/posts"); // or whichever collection
  await expect(page.getByText("Test Generated Article")).toBeVisible();
});
```

#### 2. Scheduled Agent Execution

```typescript
// packages/cms-admin/e2e/agents/agent-scheduled.spec.ts

test("scheduled agent runs when due", async ({ page }) => {
  await mockLlmResponses(page);
  await loginAsAdmin(page);

  // 1. Configure agent with schedule
  const agentId = "content-writer";
  await page.request.patch(`/api/cms/agents/${agentId}`, {
    data: { schedule: { enabled: true, frequency: "manual" } },
  });

  // 2. Trigger scheduler manually via API
  await page.request.post("/api/cms/agents/" + agentId + "/run", {
    data: { prompt: "Generate a scheduled test article" },
  });

  // 3. Verify result in curation queue
  await navigateToCuration(page);
  await expect(page.getByText("Test Generated Article")).toBeVisible();
});
```

#### 3. Budget Enforcement

```typescript
// packages/cms-admin/e2e/agents/agent-budget.spec.ts

test("agent run blocked when budget exceeded", async ({ page }) => {
  await loginAsAdmin(page);

  // 1. Set budget to $0.01 with $0.02 already spent
  await page.request.post("/api/cms/command", {
    data: { monthlyBudgetUsd: 0.01 },
  });

  // 2. Try to run agent
  const res = await page.request.post("/api/cms/agents/content-writer/run", {
    data: { prompt: "This should be blocked" },
  });

  // 3. Expect budget error
  const body = await res.json();
  expect(body.error ?? body.status).toContain("budget");
});
```

  // 4. Verify item moved to rejected tab

});
